--Update Supplier
CREATE OR REPLACE FUNCTION update_suppliers (
  p_id int4,
  p_name text,
  p_phone text,
  p_email text,
  p_address text,
  p_is_active boolean
)
RETURNS suppliers
LANGUAGE plpgsql
AS $$
DECLARE
  new_supplier suppliers;
BEGIN 
  UPDATE suppliers
  SET 
    name = p_name,
    phone = p_phone,
    email = p_email,
    address = p_address,
    is_active = p_is_active
  WHERE
    supplier_id = p_id
    
  RETURNING * INTO new_supplier;
  RETURN new_supplier;
END;
$$

drop function handle_stock_drop;
-- Handle Stock
CREATE OR REPLACE FUNCTION handle_stock_drop(
  p_product_id int4,
  p_alert_type text,
  p_thershold_value numeric,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  -- Only insert if NO active alert exists
  IF NOT EXISTS (
    SELECT 1
    FROM stock_alerts
    WHERE product_id = p_product_id
      AND is_resolved = 'NO'
  ) THEN

    INSERT INTO stock_alerts (
      product_id,
      alert_type,
      thershold_value,
      current_value,
      is_resolved
    )
    VALUES (
      p_product_id,
      p_alert_type,
      p_thershold_value,
      p_current_value,
      'NO'
    );

  END IF;

END;
$$;

drop function resolve_stock_alerts;
-- Resolve Stock
CREATE OR REPLACE FUNCTION resolve_stock_alerts(
  p_product_id int4,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  UPDATE stock_alerts
  SET
    is_resolved = 'YES',
    current_value = p_current_value
  WHERE product_id = p_product_id
    AND is_resolved = 'NO';

END;
$$;

-- Final Stock Processing
drop function upsert_stock_alert;
CREATE OR REPLACE FUNCTION upsert_stock_alert(
  p_product_id int4,
  p_alert_type text,
  p_thershold_value numeric,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  IF p_current_value <= p_thershold_value THEN
    PERFORM handle_stock_drop(
      p_product_id,
      p_alert_type,
      p_thershold_value,
      p_current_value
    );
  ELSE
    PERFORM resolve_stock_alerts(
      p_product_id,
      p_current_value
    );
  END IF;

END;
$$;

drop function update_user_profiles;
-- update user alerts email
create or replace function update_user_profiles(
  p_user_id int4,
  p_email text
)
returns void
language plpgsql
as $$
begin
  update user_profiles
  set
    alerts_email = p_email
  where user_id = p_user_id;
end;
$$;

-- Update user password
create or replace function update_user_password(
  p_user_id int,
  p_new_password text
)
returns void
language plpgsql
as $$
begin
  update user_profiles
  set password_hash = crypt(p_new_password, gen_salt('bf'))
  where user_id = p_user_id;
end;
$$;
