create or replace function set_update_at ()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

drop trigger if exists trg_user_profiles_update_at on products;

-- user_profiles
CREATE TRIGGER trg_user_profiles_updated_at
BEFORE UPDATE ON user_profiles
FOR EACH ROW
EXECUTE FUNCTION set_update_at();

-- Function to run when Adding a Inventary Movement
CREATE OR REPLACE FUNCTION apply_inventory_movement()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.movement_type = 'ENTRADA' THEN
    UPDATE products
    SET current_stock = current_stock + NEW.quantity
    WHERE product_id = NEW.product_id;

  ELSIF NEW.movement_type = 'SALIDA' THEN
    UPDATE products
    SET current_stock = current_stock - NEW.quantity
    WHERE product_id = NEW.product_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- trigger_when_adding_Inventories
CREATE TRIGGER trg_apply_inventory
AFTER INSERT ON inventory_movements
FOR EACH ROW
EXECUTE FUNCTION apply_inventory_movement();


-- Function to Run when inserting into user_profiles
CREATE OR REPLACE FUNCTION after_user_insert()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN

  INSERT INTO customers (
    user_id,
    full_name,
    status
    )
  VALUES (
    NEW.user_id,
    NEW.full_name,
    'Active'
    );

  RETURN NEW;
END;
$$;

-- trigger_when_inserting_into_user
CREATE TRIGGER trg_after_user_insert
AFTER INSERT ON user_profiles
FOR EACH ROW
EXECUTE FUNCTION after_user_insert();
