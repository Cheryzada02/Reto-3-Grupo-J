import { createClient } from "npm:@supabase/supabase-js@2.45.4";
import nodemailer from "npm:nodemailer@6.9.16";

const cors_headers = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
  "Access-Control-Allow-Methods": "POST, OPTIONS",
};

type StockAlertWebhookPayload = {
  type: "INSERT";
  table: string;
  schema: string;
  record: {
    alert_id: number;
    product_id: number;
    alert_type: string;
    thershold_value: number;
    current_value: number;
    is_resolved: string | boolean;
    email_sent?: boolean;
    created_at?: string;
  };
  old_record: null | Record<string, unknown>;
};

function is_resolved_value(value: string | boolean | null | undefined) {
  if (typeof value === "boolean") return value;
  if (!value) return false;

  return value.toLowerCase() === "yes" || value.toLowerCase() === "true";
}

Deno.serve(async (req) => {
  try {
    if (req.method === "OPTIONS") {
      return new Response("ok", {
        headers: cors_headers,
      });
    }

    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ error: "Method not allowed" }),
        {
          status: 405,
          headers: {
            ...cors_headers,
            "Content-Type": "application/json",
          },
        },
      );
    }

    const payload = await req.json() as StockAlertWebhookPayload;
    const alert = payload.record;

    if (!alert) {
      return new Response(
        JSON.stringify({ error: "Missing webhook record" }),
        {
          status: 400,
          headers: {
            ...cors_headers,
            "Content-Type": "application/json",
          },
        },
      );
    }

    if (alert.email_sent === true) {
      return new Response(
        JSON.stringify({ success: true, skipped: "Email already sent" }),
        {
          status: 200,
          headers: {
            ...cors_headers,
            "Content-Type": "application/json",
          },
        },
      );
    }

    if (is_resolved_value(alert.is_resolved)) {
      return new Response(
        JSON.stringify({ success: true, skipped: "Alert is resolved" }),
        {
          status: 200,
          headers: {
            ...cors_headers,
            "Content-Type": "application/json",
          },
        },
      );
    }

    const supabase_url = Deno.env.get("SUPA_URL");
    const service_role_key = Deno.env.get("SUPA_SERVICE_ROLE_KEY");

    if (!supabase_url) throw new Error("Missing SUPABASE_URL");
    if (!service_role_key) throw new Error("Missing SUPABASE_SERVICE_ROLE_KEY");

    const supabase_admin = createClient(supabase_url, service_role_key);

    const username = Deno.env.get("GMAIL_USERNAME");
    const app_password = Deno.env.get("GMAIL_APP_PASSWORD");

    if (!username) throw new Error("Missing GMAIL_USERNAME");
    if (!app_password) throw new Error("Missing GMAIL_APP_PASSWORD");

    /*
      Adjust this query to your real admin table.

      Example assumption:
      user_profiles has:
        user_id
        full_name
        internal_email
        role_id
        is_active

      role_id = 1 means admin.
    */
    const { data: admins, error: admins_error } = await supabase_admin
      .from("user_profiles")
      .select("alerts_email, full_name")
      .eq("role_id", 1)
      .eq("is_active", true);

    if (admins_error) {
      throw new Error(admins_error.message);
    }

    const admin_emails = admins
      ?.map((admin) => admin.alerts_email)
      .filter(Boolean);

    if (!admin_emails || admin_emails.length === 0) {
      throw new Error("No active admin emails found.");
    }

    /*
      Optional: get product info.
      Adjust column names to your products table.
    */
    const { data: product } = await supabase_admin
      .from("products")
      .select("product_name, current_stock, min_stock")
      .eq("product_id", alert.product_id)
      .single();

    const product_name = product?.name ?? `Producto ID ${alert.product_id}`;

    const transporter = nodemailer.createTransport({
      host: "smtp.gmail.com",
      port: 465,
      secure: true,
      auth: {
        user: username,
        pass: app_password,
      },
    });

    const format_date = (value: string | null | undefined) => {
      if (!value) return "";

      return new Intl.DateTimeFormat("es-DO", {
        dateStyle: "medium",
        timeStyle: "short",
        timeZone: "America/Santo_Domingo",
      }).format(new Date(value));
    };

    const subject = `Alerta de inventario: ${product_name}`;

    const html = `
      <h2>Alerta de inventario</h2>

      <p>Se ha generado una alerta de stock que requiere atención.</p>

      <table border="1" cellpadding="8" cellspacing="0">
        <tr>
          <td><strong>Producto</strong></td>
          <td>${product_name}</td>
        </tr>
        <tr>
          <td><strong>Producto ID</strong></td>
          <td>${alert.product_id}</td>
        </tr>
        <tr>
          <td><strong>Tipo de alerta</strong></td>
          <td>${alert.alert_type}</td>
        </tr>
        <tr>
          <td><strong>Stock actual</strong></td>
          <td>${alert.current_value}</td>
        </tr>
        <tr>
          <td><strong>Valor mínimo / límite</strong></td>
          <td>${alert.thershold_value}</td>
        </tr>
        <tr>
          <td><strong>Fecha</strong></td>
          <td>${format_date(alert.created_at)}</td>
        </tr>
      </table>
    `;

    const info = await transporter.sendMail({
      from: username,
      to: admin_emails.join(","),
      subject,
      html,
    });

    const { error: update_error } = await supabase_admin
      .from("stock_alerts")
      .update({
        email_sent: true,
        email_sent_at: new Date().toISOString(),
        email_error: null,
        email_attempts: 1,
      })
      .eq("alert_id", alert.alert_id);

    if (update_error) {
      throw new Error(update_error.message);
    }

    return new Response(
      JSON.stringify({
        success: true,
        message: "Stock alert email sent",
        message_id: info.messageId,
        sent_to: admin_emails,
      }),
      {
        status: 200,
        headers: {
          ...cors_headers,
          "Content-Type": "application/json",
        },
      },
    );
  } catch (error) {
    console.error("send-stock-alert-email error:", error);

    return new Response(
      JSON.stringify({
        error: error instanceof Error ? error.message : "Unknown error",
      }),
      {
        status: 500,
        headers: {
          ...cors_headers,
          "Content-Type": "application/json",
        },
      },
    );
  }
});
