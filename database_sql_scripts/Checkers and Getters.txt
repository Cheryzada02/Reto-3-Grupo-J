--Check if role exists
CREATE OR REPLACE FUNCTION check_role_exists(p_role_id int)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM roles WHERE role_id = p_role_id) THEN
    RAISE EXCEPTION 'Role No Existe';
  END IF;
END;
$$;

--Check if Supplier exists
CREATE OR REPLACE FUNCTION check_supplier_exists(p_supplier_id int)

drop function check_stock_available;
--Check stock availability
CREATE OR REPLACE FUNCTION check_stock_available(p_product_id int, p_qty numeric)
RETURNS void
LANGUAGE plpgsql
AS $$
DECLARE
  v_stock int;
BEGIN
  SELECT current_stock INTO v_stock
  FROM products
  WHERE product_id = p_product_id;

  IF v_stock < p_qty THEN
    RAISE EXCEPTION 'No Hay Stock Suficiente';
  END IF;
END;
$$;


drop function login_user;
--Validate Login
create or replace function login_user(
  p_email text,
  p_password text
)
returns json
language plpgsql
as $$
declare
  stored_hash text;
  v_user_id int;
  v_role_id int;
  v_user_name text;
begin
  select user_id, password_hash, role_id, full_name
  into v_user_id, stored_hash, v_role_id, v_user_name
  from user_profiles
  where internal_email = p_email;

  if stored_hash is null then
    return null;
  end if;

  if crypt(p_password, stored_hash) = stored_hash then
    if v_role_id != 1 THEN
      update customers 
      set last_login = now() 
      where user_id = v_user_id;
    end if;
    
    return json_build_object(
      'user_id', v_user_id,
      'role_id', v_role_id,
      'user_name', v_user_name
    );
    
    

  else
    return null;
  end if;
end;
$$;

drop function get_customer_info;
-- Get_Customer_Info
CREATE OR REPLACE FUNCTION get_customer_info(
  p_user_id int4
)
RETURNS JSON
LANGUAGE plpgsql
AS $$
DECLARE
  result JSON;
BEGIN
  
  SELECT row_to_json(vc)
  INTO result
  FROM view_customers vc
  WHERE vc.user_id = p_user_id;

  RETURN result;

END;
$$;

drop function handle_stock_drop;
-- Handle Stock
CREATE OR REPLACE FUNCTION handle_stock_drop(
  p_product_id int4,
  p_alert_type text,
  p_thershold_value numeric,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  -- Only insert if NO active alert exists
  IF NOT EXISTS (
    SELECT 1
    FROM stock_alerts
    WHERE product_id = p_product_id
      AND is_resolved = 'NO'
  ) THEN

    INSERT INTO stock_alerts (
      product_id,
      alert_type,
      thershold_value,
      current_value,
      is_resolved
    )
    VALUES (
      p_product_id,
      p_alert_type,
      p_thershold_value,
      p_current_value,
      'NO'
    );

  END IF;

END;
$$;

drop function resolve_stock_alerts;
-- Resolve Stock
CREATE OR REPLACE FUNCTION resolve_stock_alerts(
  p_product_id int4,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  UPDATE stock_alerts
  SET
    is_resolved = 'YES',
    current_value = p_current_value
  WHERE product_id = p_product_id
    AND is_resolved = 'NO';

END;
$$;

-- Final Stock Processing
drop function upsert_stock_alert;
CREATE OR REPLACE FUNCTION upsert_stock_alert(
  p_product_id int4,
  p_alert_type text,
  p_thershold_value numeric,
  p_current_value numeric
)
RETURNS void
LANGUAGE plpgsql
AS $$
BEGIN

  IF p_current_value <= p_thershold_value THEN
    PERFORM handle_stock_drop(
      p_product_id,
      p_alert_type,
      p_thershold_value,
      p_current_value
    );
  ELSE
    PERFORM resolve_stock_alerts(
      p_product_id,
      p_current_value
    );
  END IF;

END;
$$;


drop function get_customer_orders;
-- Get_Customer_Orders
CREATE OR REPLACE FUNCTION get_customer_orders(
  p_customer_id int4
)
RETURNS JSON
LANGUAGE plpgsql
AS $$
DECLARE
  result JSON;
BEGIN

  SELECT COALESCE(json_agg(vc), '[]'::json)
  INTO result
  FROM view_orders vc
  WHERE vc.customer_id = p_customer_id;

  RETURN result;

END;
$$;

drop function get_customer_orders_details;
-- Get_Customer_Orders
CREATE OR REPLACE FUNCTION get_customer_orders_details(
  p_order_id int8
)
RETURNS JSONB
LANGUAGE plpgsql
AS $$
DECLARE
  result JSONB;
BEGIN

  SELECT COALESCE(
    jsonb_agg(to_jsonb(vc) ORDER BY vc.order_item_id ASC),
    '[]'::jsonb
  )
  INTO result
  FROM view_orders_details vc
  WHERE vc.order_id = p_order_id;

  RETURN result;

END;
$$;
