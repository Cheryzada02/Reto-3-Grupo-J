DROP TABLE IF EXISTS roles;
CREATE TABLE roles (
  role_id int4 PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  name text NOT NULL UNIQUE,
  description text,
  created_at timestamptz DEFAULT now()
);

DROP TABLE IF EXISTS user_profiles;
CREATE TABLE user_profiles (
  user_id int4 PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  full_name text NOT NULL,
  internal_email text NOT NULL UNIQUE,
  password_hash text NOT NULL,
  role_id int4 NOT NULL,
  is_active boolean NOT NULL,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  CONSTRAINT fk_user_role
    FOREIGN KEY (role_id)
    REFERENCES roles(role_id)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);

DROP TABLE IF EXISTS suppliers;
CREATE TABLE suppliers (
  supplier_id int4 PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  name text NOT NULL,
  phone text,
  email text NOT NULL UNIQUE,
  address text,
  is_active boolean NOT NULL,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

DROP TABLE IF EXISTS products;
CREATE TABLE products (
  product_id int4 PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  name text NOT NULL,
  description text,
  supplier_id int4 NOT NULL,
  cost_price numeric NOT NULL,
  sale_price numeric NOT NULL,
  current_stock int4 NOT NULL,
  min_stock int4 NOT NULL,
  max_stock int4,
  status text NOT NULL,
  image_url text,
  department_id int4 NOT NULL,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  CONSTRAINT fk_products_supplier_id
    FOREIGN KEY (supplier_id)
    REFERENCES suppliers(supplier_id)
    ON UPDATE CASCADE
    ON DELETE RESTRICT,
  CONSTRAINT fk_products_department_id
    FOREIGN KEY (department_id)
    REFERENCES deparments(department_id)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);
